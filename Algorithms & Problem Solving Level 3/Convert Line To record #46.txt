vector <string> vFun(string infos,string delim)
{
	vector <string> vWord;
	int pos = 0; 
	string word = "";

	while ((pos = infos.find(delim)) != std::string::npos)
	{
		word = infos.substr(0, pos);
		if (word != " ")
		{
			vWord.push_back(word);
		}
		infos.erase(0, pos + delim.length());
	}
	if (infos != " ")
	{
		vWord.push_back(infos);
	}
	return vWord;
}
