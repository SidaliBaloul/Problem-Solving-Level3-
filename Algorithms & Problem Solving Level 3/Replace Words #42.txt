string Replace(string word,string replace,string newe)
{
	short pos = 0;

	while ((pos = word.find(replace)) != std::string::npos)
	{
		word = word.replace(pos, replace.length(), newe);
		pos = word.find(replace);
	}
	
	
	return word;
}
