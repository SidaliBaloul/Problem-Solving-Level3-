{

	return num;

void FillFibonacci(int num)
{
	int p1 = 0;
	int p2 = 1;
	int fb = 0;
	cout << " 1   ";
	
	for (int i = 2; i <= num; i++)
	{
		fb = p1 + p2;
		cout << fb << "   ";
		p1 = p2;
		p2 = fb;
	}
}
