string After(stInfo inf, string delim = "//")
{
	string word = "";

	word += inf.accountnumber + delim;
	word += inf.name + delim;
	word += inf.phone + delim;
	word += inf.pincode + delim;
	word += to_string(inf.accountbalance);

	return word;
}


void AddClients(string filename,string record)
